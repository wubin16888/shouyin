   164→   164→function shade(hex: string, percent: number): string {
   165→   165→  const m = hex.replace("#", "");
   166→   166→  if (m.length !== 6) return hex;
   167→   167→  const r = parseInt(m.slice(0, 2), 16);
   168→   168→  const g = parseInt(m.slice(2, 4), 16);
   169→   169→  const b = parseInt(m.slice(4, 6), 16);
   170→   170→  const f = (c: number) =>
   171→   171→    Math.max(0, Math.min(255, Math.round(c + (percent / 100) * 255)));
   172→   172→  const toHex = (c: number) => f(c).toString(16).padStart(2, "0");
   173→   173→  return `#${toHex(r)}${toHex(g)}${toHex(b)}`;
   174→   174→}

   176→   176→function parseFlavors(flavors: string | null): Array<{ category: string; flavor: string }> {
   177→   177→  if (!flavors) return [];
   178→   178→  try {
   179→   179→    const parsed = JSON.parse(flavors);
   180→   180→    return Array.isArray(parsed) ? parsed : [];
   181→   181→  } catch {
   182→   182→    return [];
   183→   183→  }
   184→   184→}

   186→   186→function flavorText(flavors: string | null): string {
   187→   187→  const list = parseFlavors(flavors);
   188→   188→  if (list.length === 0) return "";
