   262→   262→function BusinessParamsTab() {
   263→   263→  const [configs, setConfigs] = useState<SysConfigInfo[]>([]);
   264→   264→  const [loading, setLoading] = useState(true);
   265→   265→  const [editing, setEditing] = useState<SysConfigInfo | null>(null);
   266→   266→  const { toast } = useToast();
   267→   267→
   268→   268→  const load = useCallback(async () => {
   269→   269→    setLoading(true);
   270→   270→    try {
   271→   271→      setConfigs(await api.getSysConfigs("system"));
   272→   272→    } catch (e) {
   273→   273→      toast({ title: "加载失败", description: String(e), variant: "destructive" });
   274→   274→    } finally {
   275→   275→      setLoading(false);
   276→   276→    }
   277→   277→  }, [toast]);
   278→   278→
   279→   279→  useEffect(() => {
   280→   280→    load();
   281→   281→  }, [load]);
   282→   282→
   283→   283→  const cfgMap = useMemo(() => {
   284→   284→    const m = new Map<string, SysConfigInfo>();
   285→   285→    configs.forEach((c) => m.set(c.configKey, c));
   286→   286→    return m;
   287→   287→  }, [configs]);
   288→   288→
   289→   289→  const currency = cfgMap.get("currency_symbol")?.configValue ?? "¥";
   290→   290→  const businessHours = useMemo(() => {
   291→   291→    const raw = cfgMap.get("business_hours")?.configValue;
   292→   292→    if (!raw) return { open: "18:00", close: "02:00" };
   293→   293→    try {
   294→   294→      return JSON.parse(raw) as { open: string; close: string };
   295→   295→    } catch {
   296→   296→      return { open: "18:00", close: "02:00" };
   297→   297→    }
   298→   298→  }, [cfgMap]);
   299→   299→
   300→   300→  const setCurrency = async (sym: string) => {
   301→   301→    if (sym === currency) return;
   302→   302→    try {
   303→   303→      await api.updateSysConfig("currency_symbol", sym);
   304→   304→      toast({ title: "货币符号已切换", description: `所有金额展示将使用 ${sym}` });
   305→   305→      load();
   306→   306→    } catch (e) {
   307→   307→      toast({ title: "切换失败", description: String(e), variant: "destructive" });
   308→   308→    }
   309→   309→  };
   310→   310→
   311→   311→  const saveBusinessHours = async (open: string, close: string) => {
   312→   312→    try {
   313→   313→      await api.updateSysConfig("business_hours", JSON.stringify({ open, close }));
   314→   314→      toast({ title: "营业时间已更新", description: `${open} - ${close}` });
   315→   315→      load();
   316→   316→    } catch (e) {
   317→   317→      toast({ title: "更新失败", description: String(e), variant: "destructive" });
   318→   318→    }
   319→   319→  };
   320→   320→
   321→   321→  // 关键参数（除货币与营业时间，这两个有专属 UI）
   322→   322→  const otherParams = configs.filter(
   323→   323→    (c) => !["currency_symbol", "business_hours"].includes(c.configKey),
   324→   324→  );
   325→   325→
   326→   326→  return (
   327→   327→    <div className="space-y-4 pb-6">
   328→   328→      {/* 货币符号 + 营业时间 双卡片 */}
   329→   329→      <div className="grid gap-4 lg:grid-cols-2">
   330→   330→        {/* 货币符号切换 */}
   331→   331→        <DarkCard>
   332→   332→          <CardHeader className="pb-3">
   333→   333→            <div className="flex items-center gap-2">
   334→   334→              <div className="rounded-md bg-amber-500/15 p-1.5">
   335→   335→                <span className="text-amber-400 font-bold text-sm">¥</span>
   336→   336→              </div>
   337→   337→              <div>
   338→   338→                <CardTitle className="text-base text-slate-100">货币符号</CardTitle>
   339→   339→                <CardDescription className="text-slate-400 text-xs">
   340→   340→                  切换全系统金额展示符号
   341→   341→                </CardDescription>
   342→   342→              </div>
   343→   343→            </div>
   344→   344→          </CardHeader>
   345→   345→          <CardContent>
   346→   346→            <div className="grid grid-cols-2 gap-3">
   347→   347→              <button
   348→   348→                onClick={() => setCurrency("¥")}
   349→   349→                className={cn(
   350→   350→                  "rounded-xl border-2 p-5 text-center transition-all hover:scale-[1.02]",
   351→   351→                  currency === "¥"
   352→   352→                    ? "border-emerald-500 bg-emerald-500/15 shadow-lg shadow-emerald-500/20"
   353→   353→                    : "border-slate-700 bg-slate-800/50 hover:border-slate-600",
   354→   354→                )}
   355→   355→              >
   356→   356→                <div className={cn("text-4xl font-bold", currency === "¥" ? "text-emerald-300" : "text-slate-400")}>¥</div>
   357→   357→                <div className="mt-1 text-xs text-slate-400">人民币</div>
   358→   358→                {currency === "¥" && (
   359→   359→                  <Badge className="mt-2 bg-emerald-500/20 text-emerald-300 border-emerald-500/30">
   360→   360→                    <Check className="h-3 w-3 mr-1" /> 已启用
   361→   361→                  </Badge>
   362→   362→                )}
   363→   363→              </button>
   364→   364→              <button
   365→   365→                onClick={() => setCurrency("$")}
   366→   366→                className={cn(
   367→   367→                  "rounded-xl border-2 p-5 text-center transition-all hover:scale-[1.02]",
   368→   368→                  currency === "$"
   369→   369→                    ? "border-emerald-500 bg-emerald-500/15 shadow-lg shadow-emerald-500/20"
   370→   370→                    : "border-slate-700 bg-slate-800/50 hover:border-slate-600",
   371→   371→                )}
   372→   372→              >
   373→   373→                <div className={cn("text-4xl font-bold", currency === "$" ? "text-emerald-300" : "text-slate-400")}>$</div>
   374→   374→                <div className="mt-1 text-xs text-slate-400">美元</div>
   375→   375→                {currency === "$" && (
   376→   376→                  <Badge className="mt-2 bg-emerald-500/20 text-emerald-300 border-emerald-500/30">
   377→   377→                    <Check className="h-3 w-3 mr-1" /> 已启用
   378→   378→                  </Badge>
   379→   379→                )}
   380→   380→              </button>
   381→   381→            </div>
   382→   382→          </CardContent>
   383→   383→        </DarkCard>
   384→   384→
   385→   385→        {/* 营业时间 */}
   386→   386→        <DarkCard>
   387→   387→          <CardHeader className="pb-3">
   388→   388→            <div className="flex items-center gap-2">
   389→   389→              <div className="rounded-md bg-sky-500/15 p-1.5">
   390→   390→                <Clock className="h-4 w-4 text-sky-400" />
   391→   391→              </div>
   392→   392→              <div>
   393→   393→                <CardTitle className="text-base text-slate-100">营业时间</CardTitle>
   394→   394→                <CardDescription className="text-slate-400 text-xs">
   395→   395→                  跨日营业场景支持（如 18:00 - 次日 02:00）
   396→   396→                </CardDescription>
   397→   397→              </div>
   398→   398→            </div>
   399→   399→          </CardHeader>
   400→   400→          <CardContent>
   401→   401→            <BusinessHoursEditor

   410→   410→      {/* 其他参数列表 */}
   411→   411→      <DarkCard>
   412→   412→        <CardHeader className="pb-3">

   925→   925→function InlineEdit({ value, onSave, format }: {
   926→   926→  value: number;
   927→   927→  onSave: (v: number) => void;
   928→   928→  format: (v: number) => string;
   929→   929→}) {

   939→   939→      <Input

   945→   945→          setEditing(false);
   946→   946→          const n = Number(draft);
   947→   947→          if (!isNaN(n) && n !== value) onSave(n);

   952→   952→            setDraft(String(value));
   953→   953→            setEditing(false);
   954→   954→          }
   955→   955→        }}
   956→   956→        className="h-7 w-20 bg-slate-900/60 border-slate-700 text-slate-100 text-right"
   957→   957→      />
   958→   958→    );
   959→   959→  }
   960→   960→  return (
   961→   961→    <button
   962→   962→      onClick={() => setEditing(true)}
   963→   963→      className="font-mono text-slate-200 hover:text-emerald-300 hover:bg-slate-700/40 rounded px-1.5 py-0.5 transition-colors"
   964→   964→    >
   965→   965→      {format(value)}
   966→   966→    </button>
   967→   967→  );
   968→   968→}
